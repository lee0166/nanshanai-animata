static clean(text: string): string { let cleaned = text; // 1. 统一换行符 cleaned = cleaned.replace(/\r\n/g, '\n'); cleaned = cleaned.replace(/\r/g, '\n'); // 2. 去除控制字符 cleaned = cleaned.replace(/[\x00-\x08\x0B\x0C\x0E-\x1F\x7F-\x9F]/g, ''); // 3. 合并多余空行 cleaned = cleaned.replace(/\n{3,}/g, '\n\n'); // 4. 去除首尾空白 cleaned = cleaned.trim(); return cleaned; }
